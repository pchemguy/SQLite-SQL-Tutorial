JOIN
pragma_foreign_key_list
 (tables.table_name) 
tables AS ( 
    SELECT tbl_name AS table_name, sql 
    FROM sqlite_master 
    WHERE type = 'table' 
      AND name NOT LIKE 'sqlite_%' 
) 
columns AS ( 
    SELECT table_name, cid, name  AS col_name, 
           type, "notnull", dflt_value, pk 
    FROM tables AS t, 
         pragma_foreign_key_list (t.table_name) 
    ORDER BY src_table, fk_id, fk_seq 
) 
columns AS ( SELECT table_name AS src_table, "from" AS src_col, "table" AS dst_table, "to" AS dst_col, on_update, on_delete, id AS fk_id, seq AS fk_seq FROM tables AS t, pragma_foreign_key_list(t.table_name) ORDER BY src_table, fk_id, fk_seq ), columns AS ( SELECT table_name, cid, name AS col_name, type, "notnull", dflt_value, pk FROM tables AS t, pragma_table_xinfo(t.table_name) ORDER BY table_name, cid )
